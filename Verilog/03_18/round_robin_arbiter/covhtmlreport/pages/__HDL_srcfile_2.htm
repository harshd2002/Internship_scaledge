1  //design - round robin arbiter
     2  //Arbiter serves every requet for 1 clock cycle
     3  //Asynchronus active low reset
     4  
     5  //1. Arbiter will grant 1st grant as per the priority if multiple requests arrive
     6  //2. After 1st request is served, arbiter will serve another request available.
     7  
     8  `timescale 1ns / 1ps
     9  
     10  module rra #(parameter REQS = 4)
     11  	(input clk,rst,
     12  	input [REQS-1'b1:0] req,
     13  	output [REQS-1'b1:0] grant
     14  	);
     15  
     16  	parameter s0 = 3'b000,
     17  		  	s1 = 3'b001,
     18  		  	s2 = 3'b010,
     19  		  	s3 = 3'b011,
     20  		  	s4 = 3'b100;
     21  	  
     22  	  reg [$clog2(REQS) : 0] crt;
     23  	  reg [REQS-1'b1:0] gnt;
     24  
     25  	  assign grant = gnt;
     26  	  
     27  	  always @(posedge clk or negedge rst) begin
     28  	
     29  		if(!rst) begin
     30  			crt <= s0;
     31  			gnt <= 4'b0;
     32  		end
     33  		else begin
     34  
     35  			case (crt)
     36  				s0: begin
     37  					gnt <= 4'b0000;
     38  					if(req[0])
     39  						crt <= s1;
     40  					else if(req[1])
     41  						crt <= s2;
     42  					else if(req[2])
     43  						crt <= s3;
     44  					else if(req[3])
     45  						crt <= s4;
     46  					else
     47  						crt <= s0;
     48  				end
     49  				s1: begin
     50  					gnt <= 4'b0001;
     51  					if(req[1])
     52  						crt <= s2;
     53  					else if(req[2])
     54  						crt <= s3;
     55  					else if(req[3])
     56  						crt <= s4;
     57  					else if(req[0])
     58  						crt <= s1;
     59  					else
     60  						crt <= s0;
     61  				end
     62  				s2: begin
     63  					gnt <= 4'b0010;
     64  					if(req[2])
     65  						crt <= s3;
     66  					else if(req[3])
     67  						crt <= s4;
     68  					else if(req[0])
     69  						crt <= s1;
     70  					else if(req[1])
     71  						crt <= s2;
     72  					else
     73  						crt <= s0;
     74  				end
     75  				s3: begin
     76  					gnt <= 4'b0100;
     77  					if(req[3])
     78  						crt <= s4;
     79  					else if(req[0])
     80  						crt <= s1;
     81  					else if(req[1])
     82  						crt <= s2;
     83  					else if(req[2])
     84  						crt <= s3;
     85  					else
     86  						crt <= s0;
     87  				end
     88  				s4: begin
     89  					gnt <= 4'b1000;
     90  					if(req[0])
     91  						crt <= s1;
     92  					else if(req[1])
     93  						crt <= s2;
     94  					else if(req[2])
     95  						crt <= s3;
     96  					else if(req[3])
     97  						crt <= s4;
     98  					else
     99  						crt <= s0;
     100  				end
     101  				default: begin
     102  					gnt <= 4'b0;
     103  					crt <= s0;
     104  				end
     105  			endcase
     106  
     107  		end
     108  	
     109  	end
     110  
     111  endmodule
