1  //testbench - round robin arbiter
     2  //testinputs:
     3  //1. Initial reset and reset on the fly.
     4  //2. sanity - individual requests.
     5  //3. checking for multile requests.
     6  //4. checking every transition for the fsm
     7  
     8  module tb;
     9  
     10  	parameter REQS = 4;
     11  
     12  	reg clk, rst;
     13  	reg [REQS-1'b1 : 0] req;
     14  	wire [REQS - 1'b1 : 0] act_grant;
     15  
     16  	rra #(.REQS(REQS)) dut(.clk(clk), .rst(rst), .req(req), .grant(act_grant));
     17  
     18  	task reset(input integer del);
     19  	begin
     20  		@(negedge clk)
     21  		rst = 1'b0;
     22        	req = 4'b0;
     23  		repeat(del) @(negedge clk);
     24  		rst = 1'b1;
     25  	end
     26  	endtask
     27  
     28  	task request(input [REQS - 1'b1 : 0] request);
     29  	begin
     30  		@(negedge clk)
     31  		req = request;
     32  	end
     33  	endtask
     34  
     35  	initial begin
     36        
     37  		//initial reset
     38        	reset(1);
     39  
     40  		//sanity - generating individual requests
     41  		if($test$plusargs("sanity")) begin
     42  			request(1);
     43  			request(2);
     44  			request(4);
     45  			request(8);
     46  			request(0);
     47  			repeat(2) @(negedge clk);
     48  		end
     49  
     50  		//priority check when all requests arrive together and it will implement all requests one by one 
     51  		if($test$plusargs("priority_check")) begin
     52        	reset(1);
     53  		request(15);
     54        	repeat(8) @(negedge clk);
     55  		request(0);
     56        	repeat(4) @(negedge clk);
     57  		end
     58  
     59  //FSM transition check
     60  
     61  		//Transitions from s0
     62  		if($test$plusargs("ideal_transition")) begin
     63        	reset(1);
     64  		request(1);
     65  		request(0);
     66  		request(2);
     67  		request(0);
     68  		request(4);
     69  		request(0);
     70  		request(8);
     71  		request(0);
     72        	repeat(2) @(negedge clk);
     73  		end
     74  
     75  		//Transitions from s1
     76  		if($test$plusargs("s1_transition")) begin
     77        	reset(1);
     78  		request(1);
     79  		request(0);
     80  		request(3);
     81        	repeat(2) @(negedge clk);
     82  		request(5);
     83        	repeat(1) @(negedge clk);
     84  		request(9);
     85  		request(1);
     86        	repeat(1) @(negedge clk);
     87  		end
     88  
     89  		//Transitions from s2
     90  		if($test$plusargs("s2_transition")) begin
     91        	reset(1);
     92  		request(2);
     93  		request(0);
     94  		request(2);
     95  		request(3);
     96  		request(6);
     97  		request(10);
     98  		request(2);
     99        	repeat(2) @(negedge clk);
     100  		end
     101  
     102  		//Transitions from s3
     103  		if($test$plusargs("s3_transition")) begin
     104        	reset(1);
     105  		request(4);
     106  		request(0);
     107  		request(4);
     108  		request(5);
     109  		request(4);
     110  		request(6);
     111  		request(12);
     112        	repeat(1) @(negedge clk);
     113  		request(4);
     114        	repeat(2) @(negedge clk);
     115  		end
     116  
     117  		//Transitions from s4
     118  		if($test$plusargs("s4_transition")) begin
     119        	reset(1);
     120  		request(8);
     121  		request(0);
     122  		request(8);
     123  		request(9);
     124  		request(8);
     125  		request(10);
     126  		request(8);
     127  		request(12);
     128  		request(8);
     129        	repeat(2) @(negedge clk);
     130  		end
     131  
     132  		//inbetween reset (reset on the fly)
     133  		if($test$plusargs("inbetween_reset")) begin
     134        	reset(1);
     135  		request(13);
     136        	repeat(5) @(negedge clk);
     137  		request(0);
     138        	repeat(2) @(negedge clk);
     139  		end
     140  
     141        	repeat(5) @(negedge clk);
     142  		$stop;
     143  	end
     144  
     145  	initial begin
     146  	       clk = 1'b0;
     147         	   forever #5 clk = ~clk;
     148      end
     149  
     150  	initial begin
     151  		$dumpfile("dump.vcd");
     152  		$dumpvars;
     153  	end
     154  
     155  //reference model for self checking
     156  	reg [REQS-1'b1:0] exp_grant, gnt, flag;
     157  
     158  	always @(posedge clk or negedge rst) begin
     159  		
     160  		exp_grant <= gnt;
     161  		if(!rst) begin
     162  			exp_grant <= 4'h0;
     163  			gnt <= 4'h0;
     164  			flag <=4'h0;
     165  		end
     166  		else begin
     167  
     168  		if(req[0] && !flag[0]) begin
     169  			gnt <= 4'h1;
     170  			flag <= 4'h1;
     171  			if(req[REQS-1'b1:1] == 0) 
     172  				flag <= 4'h0;
     173  		end
     174  		else if(req[1] && !flag[1]) begin
     175  			gnt <= 4'h2;
     176  			flag <= 4'h3;
     177  			if(req[REQS-1'b1:2] == 0) 
     178  				flag <= 4'h0;
     179  		end
     180  		else if(req[2] && !flag[2]) begin
     181  			gnt <= 4'h4;
     182  			flag <= 4'h7;
     183  			if(req[3] == 0) 
     184  				flag <= 4'h0;
     185  		end
     186  		else if(req[3] && !flag[3]) begin
     187  			gnt <= 4'h8;
     188  			flag <= 4'h0;
     189  		end
     190  		else begin
     191  			gnt <= 4'h0;
     192  			flag <= 4'h0;
     193  		end
     194  		end
     195  
     196  	end
     197  
     198  //checker
     199  	initial begin
     200  		forever @(exp_grant or act_grant) begin
     201  		#1;					//due to delta simulation gap, signals may differ while comparing. So delay of 1 was given to get stable signals
     202  
     203  		$display("--------------------------------------------------------------");
     204  		$display($time," :request = %b, actual grant = %b, expected grant = %b", req, act_grant, exp_grant);
     205  			if(exp_grant == act_grant)
     206  				$display($time," :grant is passed.");
     207  			else
     208  				$display($time," :grant is not passed.");
     209  		$display("--------------------------------------------------------------");
     210  		end
     211  	end
     212  
     213  endmodule
