     3  module tb();
     4  
     5  	parameter WIDTH = 8,
     6  			DEPTH = 16,
     7          	SIZE = $clog2(DEPTH),
     8  			DIFFERENCE = 2;
     9  	
     10  	reg wclk, rclk, wrstn, rrstn, wen, ren;
     11  	reg [WIDTH-1'b1 : 0] wdata;
     12  	wire [WIDTH-1'b1 : 0] act_rdata;
     13  	wire act_full, act_empty, act_almost_full, act_almost_emp;
     14  	
     15  	async_fifo #(.WIDTH(WIDTH),
     16  				.DEPTH(DEPTH),
     17  				.SIZE(SIZE),
     18  				.DIFF(DIFFERENCE))
     19  				dut(.wr_clk(wclk),
     20  					.rd_clk(rclk),
     21  					.wr_rstn(wrstn),
     22  					.rd_rstn(rrstn),
     23  					.wr_en(wen),
     24  					.rd_en(ren),
     25  					.wr_data(wdata),
     26  					.rd_data(act_rdata),
     27  					.full(act_full),
     28  					.empty(act_empty),
     29  					.almost_full(act_almost_full),
     30  					.almost_empty(act_almost_empty)
     31  				);
     32  
     33  	task wr_resetn();
     34  	begin
     35  		@(negedge wclk);
     36  		wrstn = 1'b0;
     37  		wen = 1'b0;
     38  		wdata = 'd0;
     39  		@(posedge wclk);
     40  		wrstn = 1'b1;
     41  	end
     42  	endtask
     43  
     44  	task rd_resetn();
     45  	begin
     46  		@(negedge rclk);
     47  		rrstn = 1'b0;
     48  		ren = 1'b0;
     49  		@(posedge rclk);
     50  		rrstn = 1'b1;
     51  	end
     52  	endtask
     53  
     54  	task write(input reg [WIDTH-1'b1 : 0] data);
     55  		begin
     56  			@(negedge wclk);
     57  			wdata = data;
     58  			wen = 1'b1;
     59  		end
     60  	endtask
     61  
     62  	task read();
     63  		begin
     64  			@(negedge rclk);
     65  			ren = 1'b1;
     66  		end
     67  	endtask
     68  
     69  	task wr_clock_delay(input integer del);
     70  		repeat(del) @(posedge wclk);
     71  	endtask
     72  
     73  	task rd_clock_delay(input integer del);
     74  		repeat(del) @(posedge rclk);
     75  	endtask
     76  
     77  	initial begin
     78  		fork
     79  		wr_resetn();
     80  		rd_resetn();
     81  		join
     82  			@(negedge rclk);
     83  			ren = 1'b0;
     84  			wen = 1'b0;
     85  
     86  //sanity check
     87  		if($test$plusargs("sanity_test")) begin
     88  		write(23);
     89  		write(56);
     90  			@(negedge rclk);
     91  			wen = 1'b0;
     92  		read();
     93  		read();
     94  			@(negedge rclk);
     95  			ren = 1'b0;
     96  		end
     97  
     98  //full flag check by completely reading the fifo
     99  		if($test$plusargs("fifo_full")) begin
     100  		fork
     101  		wr_resetn();
     102  		rd_resetn();
     103  		join
     104  		repeat(DEPTH) begin
     105  			write({$random});
     106  		end
     107  			@(negedge rclk);
     108  			wen = 1'b0;
     109  		repeat(DEPTH) begin
     110  			read();
     111  		end
     112  			@(negedge rclk);
     113  			ren = 1'b0;
     114  		repeat(DEPTH) begin
     115  			write({$random});
     116  		end
     117  			@(negedge rclk);
     118  			wen = 1'b0;
     119  		end
     120  
     121  //full flag check by half reading the fifo
     122  		if($test$plusargs("fifo_half_full")) begin
     123  		fork
     124  		wr_resetn();
     125  		rd_resetn();
     126  		join
     127  		repeat(DEPTH) begin
     128  			write({$random});
     129  		end
     130  			@(negedge rclk);
     131  			wen = 1'b0;
     132  		repeat(DEPTH/2) begin
     133  			read();
     134  		end
     135  			@(negedge rclk);
     136  			ren = 1'b0;
     137  		repeat(DEPTH) begin
     138  			write({$random});
     139  		end
     140  			@(negedge rclk);
     141  			wen = 1'b0;
     142  		end
     143  
     144  //empty flag check
     145  		if($test$plusargs("fifo_empty")) begin
     146  		fork
     147  		wr_resetn();
     148  		rd_resetn();
     149  		join
     150  		repeat(DEPTH) begin
     151  			write({$random});
     152  		end
     153  			@(negedge rclk);
     154  			wen = 1'b0;
     155  		repeat(DEPTH) begin
     156  			read();
     157  		end
     158  			@(negedge rclk);
     159  			ren = 1'b0;
     160  		repeat(DEPTH) begin
     161  			write({$random});
     162  		end
     163  			@(negedge rclk);
     164  			wen = 1'b0;
     165  		repeat(DEPTH) begin
     166  			read();
     167  		end
     168  			@(negedge rclk);
     169  			ren = 1'b0;
     170  		end
     171  
     172  //simultaneous read write when fifo is about to get full to see if it hangs or not
     173  		if($test$plusargs("simultaneous_rw")) begin
     174  		fork
     175  		wr_resetn();
     176  		rd_resetn();
     177  		join
     178  		repeat(DEPTH - 2'd2) begin
     179  			write({$random});
     180  		end
     181  			@(negedge rclk);
     182  			wen = 1'b0;
     183  		repeat(5) fork
     184  			write({$random});
     185  			read();
     186  		join
     187  			@(negedge rclk);
     188  			ren = 1'b0;
     189  			wen = 1'b0;
     190  		end
     191  
     192  //continuous read write when fifo is almost full to check if fifo hangs or not
     193  		if($test$plusargs("continuous_rw")) begin
     194  		fork
     195  		wr_resetn();
     196  		rd_resetn();
     197  		join
     198  		repeat(DEPTH - 2'd2) begin
     199  			write({$random});
     200  		end
     201  			@(negedge rclk);
     202  			wen = 1'b0;
     203  		repeat(5) begin
     204  			write({$random});
     205  			read();
     206  		end
     207  			@(negedge rclk);
     208  			ren = 1'b0;
     209  			wen = 1'b0;
     210  		end
     211  
     212  		#50;
     213  		$finish;
     214  
     215  	end
     216  
     217  	initial begin
     218  		wclk = 1'b0;
     219  		forever #10 wclk = ~wclk;
     220  	end
     221  
     222  	initial begin
     223  		rclk = 1'b0;
     224  		forever #20 rclk = ~rclk;
     225  	end
     226  
     227  	initial begin
     228  		$dumpfile("dump.vcd");
     229  		$dumpvars;
     230  	end
     231  
     232  //reference model for self checking
     233      reg [WIDTH-1'b1:0] ref_mem [0 : DEPTH-1'b1];
     234      reg [WIDTH-1'b1 : 0] exp_rd_data, wr_1, wr_2, rd_1, rd_2;
     235      reg exp_full, exp_empty, exp_almost_full, exp_almost_empty;
     236  	reg exp_overflow, exp_underflow;
     237      reg [SIZE-1'b1 : 0] wr_ptr, rd_ptr;
     238  	reg wr_wrap, rd_wrap;
     239  	
     240  	always @(posedge wclk or negedge wrstn) begin
     241  		if(!wrstn) begin
     242  			wr_ptr <= 'd0;
     243  			exp_full <= 1'b0;
     244  			exp_almost_full <= 1'b0;
     245  			wr_wrap = 1'b0;
     246  		end
     247  		else if(wen && !exp_full) begin
     248  			ref_mem[wr_ptr[SIZE-1'b1 : 0]] <= wdata;
     249  			wr_ptr <= wr_ptr + 1'b1;
     250  			if(wr_ptr == (DEPTH-1'b1))
     251  			wr_wrap = ~wr_wrap;
     252  		end
     253  	end
     254  
     255  	always @(posedge rclk or negedge rrstn) begin
     256  		if(!rrstn) begin
     257  			rd_ptr <= 'd0;
     258  			exp_empty <= 1'b1;
     259  			exp_rd_data <= 'd0;
     260  			exp_almost_empty <= 1'b0;
     261  			rd_wrap = 1'b0;
     262  		end
     263  		else if(ren && !exp_empty) begin
     264  			exp_rd_data <= ref_mem[rd_ptr[SIZE-1'b1 : 0]] ;
     265  			rd_ptr <= rd_ptr + 1'b1;
     266  			if(rd_ptr == (DEPTH-1'b1))
     267  			rd_wrap = ~rd_wrap;
     268  		end
     269  	end
     270  
     271  	always @(posedge wclk or negedge wrstn) begin
     272  		if(!wrstn) begin
     273  			rd_1 <= 'd0;
     274  			rd_2 <= 'd0;
     275  		end
     276  		else begin	
     277  			rd_2 <= rd_ptr;
     278  			rd_1 <= rd_2;
     279  		end
     280  	end
     281  
     282  	always @(posedge rclk or negedge rrstn) begin
     283  		if(!rrstn) begin
     284  			wr_1 <= 'd0;
     285  			wr_2 <= 'd0;
     286  		end
     287  		else begin	
     288  			wr_2 <= wr_ptr;
     289  			wr_1 <= wr_2;
     290  		end
     291  	end
     292  
     293  	always @(wr_ptr or rd_1) begin
     294  		if((wr_wrap != rd_wrap) && (wr_ptr == rd_1)) begin
     295  			exp_full = 1'b1;
     296  		end
     297  		else begin
     298  			exp_full = 1'b0;
     299  		end
     300  
     301  		if(wr_wrap != rd_wrap) begin
     302  			if(((rd_1 - wr_ptr) <= DIFFERENCE) && (!exp_full))
     303  				exp_almost_full = 1'b1;
     304  			else
     305  				exp_almost_full = 1'b0;
     306  		end
     307  		else begin
     308  			if(((wr_ptr - rd_1) >= (DEPTH - DIFFERENCE)) && (!exp_full))
     309  				exp_almost_full = 1'b1;
     310  			else
     311  				exp_almost_full = 1'b0;
     312  		end
     313  
     314  	end
     315  
     316  	always @(rd_ptr or wr_1) begin
     317  		if((rd_ptr == wr_1) && (wr_wrap == rd_wrap)) begin
     318  			exp_empty = 1'b1;
     319  		end
     320  		else begin
     321  			exp_empty = 1'b0;
     322  		end
     323  
     324  		if(wr_wrap != rd_wrap) begin
     325  			if(((rd_ptr - wr_1) >= (DEPTH-DIFFERENCE)) && !exp_empty)
     326  				exp_almost_empty = 1'b1;
     327  			else
     328  				exp_almost_empty = 1'b0;
     329  		end
     330  		else begin
     331  			if(((wr_1 - rd_ptr) <= DIFFERENCE) && !exp_empty)
     332  				exp_almost_empty = 1'b1;
     333  			else
     334  				exp_almost_empty = 1'b0;
     335  		end
     336  	end
     337  
     338  //checker for selfchecking
     339  	initial begin
     340  
     341  		forever @(exp_rd_data, act_rdata) begin
     342  		#1;								//delay in order to get stable values so that can be compared properly
     343  			$display("--------------------------------------------------------------------");
     344  				$display($time," :read pointer = %d,write pointer = %d ",rd_ptr, wr_ptr);
     345  			if(exp_rd_data == act_rdata) begin
     346  				$display($time," :read data is passed");
     347  				$display($time," :actual read = %d, expected read = %d ",act_rdata, exp_rd_data);
     348  			end
     349  			else begin
     350  				$display($time," :read data is failed ");
     351  				$display($time," :actual read = %d, expected read = %d ",act_rdata, exp_rd_data);
     352  			end
     353  
     354  			if(exp_full == act_full) begin
     355  				$display($time," :full flag is passed");
     356  			end
     357  			else begin
     358  				$display($time," :full flag is failed ");
     359  			end
     360  			
     361  			if(exp_empty == act_empty) begin
     362  				$display($time," :empty flag is passed");
     363  			end
     364  			else begin
     365  				$display($time," :empty flag is failed ");
     366  			end
     367  			
     368  			if(exp_almost_full == act_almost_full) begin
     369  				$display($time," :almost full is passed");
     370  			end
     371  			else begin
     372  				$display($time," :almost full is failed ");
     373  			end
     374  			if(exp_almost_empty == act_almost_empty) begin
     375  				$display($time," :almost empty is passed");
     376  			end
     377  			else begin
     378  				$display($time," :almost empty is failed ");
     379  			end
     380  			$display("--------------------------------------------------------------------");
     381  
     382  		end
     383  
     384  
     385  	end
