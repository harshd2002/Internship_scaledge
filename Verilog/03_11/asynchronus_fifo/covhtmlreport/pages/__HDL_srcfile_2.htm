1  //Design - Asynchronus FIFO
     2  //
     3  
     4  `timescale 1ns / 1ps
     5  
     6  module  async_fifo #(parameter WIDTH = 8,
     7                                 DEPTH = 16,
     8                                 SIZE = $clog2(DEPTH),
     9  							   DIFF = 2
     10                      )
     11                     (input wr_clk, rd_clk,
     12                      input wr_rstn, rd_rstn,
     13                      input wr_en, rd_en,
     14                      input [WIDTH-1'b1 : 0] wr_data,
     15                      output reg [WIDTH-1'b1 : 0] rd_data,
     16  					output reg full, empty, almost_full, almost_empty,
     17  					output reg overflow, underflow
     18                      );
     19  
     20      reg [WIDTH-1'b1:0] mem [0 : DEPTH-1'b1];
     21      reg [SIZE : 0] wr_ptr, rd_ptr;
     22  	wire [SIZE : 0] g_wr_ptr, g_rd_ptr;
     23  	reg [SIZE : 0] wr_ptr_1, rd_ptr_1;
     24  	reg [SIZE : 0] wr_ptr_3, rd_ptr_3;
     25  	wire [SIZE : 0] wr_ptr_2, rd_ptr_2;
     26  
     27  	assign g_wr_ptr = binary_grey(wr_ptr);
     28  	assign rd_ptr_2 = grey_binary(rd_ptr_1);
     29  	assign g_rd_ptr = binary_grey(rd_ptr);
     30  	assign wr_ptr_2 = grey_binary(wr_ptr_1);
     31  
     32  	function reg [SIZE : 0] binary_grey(input [SIZE :0] ptr_value);
     33  	integer i;
     34  	begin
     35  		i = 0;
     36  		binary_grey[SIZE] = ptr_value[SIZE];
     37  		repeat(SIZE) begin
     38  			binary_grey[i] = (ptr_value[i] ^ ptr_value[i+1'b1]);
     39  			i = i + 1'b1;
     40  		end
     41  	end
     42  	endfunction
     43  
     44  	function reg [SIZE : 0] grey_binary(input [SIZE :0] ptr_value);
     45  	integer i; 
     46  	begin
     47  		i = SIZE-1'b1;
     48  		grey_binary[SIZE] = ptr_value[SIZE];
     49  		repeat(SIZE) begin
     50  			grey_binary[i] = (ptr_value[i] ^ grey_binary[i+1'b1]);
     51  			i = i - 1'b1;
     52  		end
     53  	end
     54  	endfunction
     55  
     56  	always @(posedge wr_clk or negedge wr_rstn) begin
     57  		if(!wr_rstn) begin
     58  			wr_ptr <= 'd0;
     59  			full <= 1'b0;
     60  			almost_full <= 1'b0;
     61  		end
     62  		else if(wr_en && !full) begin
     63  			mem[wr_ptr[SIZE-1'b1 : 0]] <= wr_data;
     64  			wr_ptr <= wr_ptr + 1'b1;
     65  		end
     66  	end
     67  
     68  	always @(posedge rd_clk or negedge rd_rstn) begin
     69  		if(!rd_rstn) begin
     70  			rd_ptr <= 'd0;
     71  			empty <= 1'b1;
     72  			rd_data <= 'd0;
     73  			almost_empty <= 1'b0;
     74  		end
     75  		else if(rd_en && !empty) begin
     76  			rd_data <= mem[rd_ptr[SIZE-1'b1 : 0]] ;
     77  			rd_ptr <= rd_ptr + 1'b1;
     78  		end
     79  	end
     80  
     81  	always @(posedge wr_clk or negedge wr_rstn) begin
     82  		if(!wr_rstn) begin
     83  			rd_ptr_1 <= 'd0;
     84  			rd_ptr_3 <= 'd0;
     85  		end
     86  		else begin	
     87  			rd_ptr_3 <= g_rd_ptr;
     88  			rd_ptr_1 <= rd_ptr_3;
     89  		end
     90  	end
     91  
     92  	always @(posedge rd_clk or negedge rd_rstn) begin
     93  		if(!rd_rstn) begin
     94  			wr_ptr_1 <= 'd0;
     95  			wr_ptr_3 <= 'd0;
     96  		end
     97  		else begin	
     98  			wr_ptr_3 <= g_wr_ptr;
     99  			wr_ptr_1 <= wr_ptr_3;
     100  		end
     101  	end
     102  
     103  	always @(wr_ptr or rd_ptr_2) begin
     104  
     105  		if(wr_en && full)
     106  			overflow = 1'b1;
     107  		else
     108  			overflow = 1'b0;
     109  
     110  		if((wr_ptr[SIZE] != rd_ptr_2[SIZE]) && (wr_ptr[SIZE-1'b1 : 0] == rd_ptr_2[SIZE-1'b1 : 0])) begin
     111  			full = 1'b1;
     112  		end
     113  		else begin
     114  			full = 1'b0;
     115  		end
     116  
     117  		if(wr_ptr[SIZE] != rd_ptr_2[SIZE]) begin
     118  			if(((rd_ptr_2[SIZE-1'b1:0] - wr_ptr[SIZE-1'b1:0]) <= DIFF) && (!full))
     119  				almost_full = 1'b1;
     120  			else
     121  				almost_full = 1'b0;
     122  		end
     123  		else begin
     124  			if(((wr_ptr[SIZE-1'b1:0] - rd_ptr_2[SIZE-1'b1:0]) >= (DEPTH - DIFF)) && (!full))
     125  				almost_full = 1'b1;
     126  			else
     127  				almost_full = 1'b0;
     128  		end
     129  
     130  	end
     131  
     132  	always @(rd_ptr or wr_ptr_2) begin
     133  
     134  		if(rd_en && empty)
     135  			underflow = 2'b1;
     136  		else
     137  			underflow = 1'b0;
     138  
     139  		if(rd_ptr == wr_ptr_2) begin
     140  			empty = 1'b1;
     141  		end
     142  		else begin
     143  			empty = 1'b0;
     144  		end
     145  
     146  		if(rd_ptr[SIZE] != wr_ptr_2[SIZE]) begin
     147  			if(((rd_ptr[SIZE-1'b1:0] - wr_ptr_2[SIZE-1'b1:0]) >= (DEPTH-DIFF)) && !empty)
     148  				almost_empty = 1'b1;
     149  			else
     150  				almost_empty = 1'b0;
     151  		end
     152  		else begin
     153  			if(((wr_ptr_2[SIZE-1'b1:0] - rd_ptr[SIZE-1'b1:0]) <= DIFF) && !empty)
     154  				almost_empty = 1'b1;
     155  			else
     156  				almost_empty = 1'b0;
     157  		end
     158  	end
     159  
     160  endmodule
